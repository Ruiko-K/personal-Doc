检查Oracle数据库被锁的对象

1、检查被锁的表
SELECT dob.object_name table_name,
       lo.session_id,
       vss.serial#,
       vss.action action,
       vss.osuser osuser,
       vss.process ap_pid,
       vps.spid db_pid,
       vss.module,
       vss.machine,
       'ALTER SYSTEM KILL SESSION ''' || vss.sid || ',' || vss.serial# || ''';'
  FROM v$locked_object lo, dba_objects dob, v$session vss, v$process vps
 WHERE lo.object_id = dob.object_id
   AND lo.session_id = vss.sid
   AND vss.paddr = vps.addr
 ORDER BY 2, 3, dob.object_name;



1、检查被锁的包
SELECT pkg.NAME,
       pkg.owner,
       acc.sid,
       ses.SERIAL#,
       ses.PADDR,
       prc.spid,
       spid,
       vss.module,
       vss.machine,
       'ALTER SYSTEM KILL SESSION '''|| SES.SID || ',' || SES.SERIAL# || ''' IMMEDIATE;'
  FROM V$DB_OBJECT_CACHE pkg,
       v$access acc,
       v$session ses,
       v$process prc
 WHERE pkg.owner = acc.owner
   AND pkg.NAME = acc.OBJECT
   AND acc.SID = ses.SID
   AND prc.addr = ses.paddr
   AND pkg.OWNER = 'APPS'
   AND pkg.LOCKS != '0'
   AND pkg.NAME LIKE 'CUX_FDPLB_RPT_PKG';
